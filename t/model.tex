\be
\rr = \rr (u^1, u^2) \quad
\e_\alpha = \rr _{,_\alpha} \quad
g_{\alpha \beta} = \e_\alpha \e_\beta \quad
\n \propto \e_1 \times \e_2 \\
H_{\alpha \beta} = \e_\alpha \n_{,\beta} \quad
H = trace(\HH) = H_{\alpha \beta} g^{\alpha \beta}
\ee

\bframe{Energy}
\be
G \propto \int H^2 \, dA, \quad M \propto \int H \, dA
\ee
